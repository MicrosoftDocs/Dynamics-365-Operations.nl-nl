        <trans-unit xml:space="preserve" translate="yes" id="106">

        <target logoport:matchpercent="101" state="translated" state-qualifier="leveraged-tm">Daarom geven ontvangers van goederen en/of diensten zowel de te betalen btw (in de rol van verkoper) en de te vorderen btw (in de rol van inkoper) in hun btw-aangifte aan.</target></trans-unit>

        <trans-unit xml:space="preserve" translate="yes" id="108">

        <target logoport:matchpercent="101" state="translated" state-qualifier="leveraged-tm">In andere landen of regio's hangt de verantwoordelijkheid voor btw-betaling af van de status van de leverancier en de koper.</target></trans-unit>
